package com.company;

import javax.imageio.ImageIO;
import java.awt.*;
import java.awt.image.BufferedImage;
import java.io.File;

public class Rgb {
    public static void main (String[] args) throws Exception{
        File F1=new File("C:\\Users\\dell\\Desktop\\rash.jpeg");
        BufferedImage img= ImageIO.read(F1);
        System.out.println("Read");
        int height= img.getHeight();
        int width=img.getWidth();

        int counter=0;
        for (int i=0;i<=height;i++){
            for(int j=0;j<=width;j++){
                counter++;
                Color c=new Color(img.getRGB(j,i));
                System.out.println("Sno:"+counter);
                System.out.println("Red:"+c.getRed());
                System.out.println("Green:"+c.getGreen());
                System.out.println("bLUE:"+c.getBlue());
            }
        }
    }
}
